% --- Boîte de Résumé / Recommandation ---
% Usage: \begin{summarybox}{Titre} ... \end{summarybox}
\newtcolorbox{summarybox}[1]{
    colback=primaryColor!5!white,
    colframe=primaryColor,
    fonttitle=\bfseries,
    title=#1,
    sharp corners=south,
    boxrule=0.5mm
}
