% =============================================================================

% Boîte de résumé ou recommandation
% Usage: \begin{summarybox}{Titre} Contenu \end{summarybox}
\newtcolorbox{summarybox}[1]{
    colback=primaryColor!5!white,
    colframe=primaryColor,
    fonttitle=\bfseries,
    title=#1,
    sharp corners=south,
    boxrule=0.5mm
}

% Boîte d'information
% Usage: \begin{infobox}{Titre} Contenu \end{infobox}
\newtcolorbox{infobox}[1]{
    colback=brandCyan!5!white,
    colframe=brandCyan,
    fonttitle=\bfseries,
    title=#1,
    sharp corners=south,
    boxrule=0.5mm
}

% Boîte d'alerte ou d'attention
% Usage: \begin{alertbox}{Titre} Contenu \end{alertbox}
\newtcolorbox{alertbox}[1]{
    colback=red!5!white,
    colframe=red!75!black,
    fonttitle=\bfseries,
    title=#1,
    sharp corners=south,
    boxrule=0.5mm
}
